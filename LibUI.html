def init(opt = nil)
  unless opt
    opt = FFI::InitOptions.malloc
    opt.to_ptr.free = Fiddle::RUBY_FREE
  end
  i = super(opt)
  return if i.size.zero?

  warn 'error'
  warn UI.free_init_error(init)
end
